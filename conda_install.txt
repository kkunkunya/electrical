# Conda安装命令
# 基础数据处理包
conda install pandas numpy pyyaml

# pydantic需要从conda-forge安装
conda install -c conda-forge pydantic

# 优化求解包
conda install -c conda-forge cvxpy
conda install -c conda-forge pulp

# 科学计算和可视化
conda install scipy matplotlib seaborn

# Jupyter支持
conda install jupyter ipykernel

# 测试框架
conda install -c conda-forge pytest pytest-cov

# 或者一次性安装所有包：
conda install pandas numpy pyyaml scipy matplotlib seaborn jupyter ipykernel -c conda-forge pydantic cvxpy pulp pytest pytest-cov
